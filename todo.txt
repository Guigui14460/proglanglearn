def send_course_link(sender, instance, created, **kwargs):
    if created:
    	all_users_course_subscription = Profile.objects.filter(new_course_email=True)
    	all_users_course_subscription_email = [user_subscription.user.email for user_subscription in all_users_course_subscription]
    	send_mail("Un nouveau cours viens d'être créer !", "Viens le voir ! " + instance.title, settings.EMAIL_HOST_USER, all_users_course_subscription_email, fail_silently=False)

post_save.connect(send_course_link, sender=Course)
