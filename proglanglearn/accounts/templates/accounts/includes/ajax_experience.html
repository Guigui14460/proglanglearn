function ajaxExperience(){$('#experience-section form').submit(function(e){e.preventDefault();$.ajax({method: $(this).attr('method'),url: $(this).attr('action'),data: $(this).serialize() + "&experience=",dataType: 'json',success: function(response){$('#experience-section').html(response.html);ajaxExperience();},error: function(rs, e){console.log(rs.responseText);},});});}ajaxExperience();
