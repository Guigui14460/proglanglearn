function ajaxEducation(){ $('#education-section form').submit(function(e){ e.preventDefault(); $.ajax({ method: $(this).attr('method'), url: $(this).attr('action'), data: $(this).serialize() + "&education=", dataType: 'json', success: function(response){ $('#education-section').html(response.html); ajaxEducation(); }, error: function(rs, e){ console.log(rs.responseText); }, }); }); } ajaxEducation();
